/* TinyC self test -- theme 3: data structures, sorting & number/string formatting.
 *
 * TinyC has no arrays or globals, so a stack is encoded inside a single int: each
 * element (a value 0..8) is stored as one base-10 digit, pushed as value+1 so that a
 * 0 digit means "empty". On top of that abstraction: push/pop/peek/size/isEmpty,
 * sum/max/contains, a counting sort, and a stack reversal (checked against its own
 * involution). Plus a 2x2 matrix multiply built from lookup functions, a binary search
 * over the virtual array f(i)=i*i, and recursive decimal / arbitrary-base printers.
 *
 * A stack holds up to 9 elements: 999999999 < 2^31-1, so the packing never overflows.
 * main() returns the number of failed checks -> exit code 0 means success. **/

func push(int s, int x) {                    // x in 0..8; store x+1 so 0 == empty slot
    return s * 10 + (x + 1);
}

func popTop(int s) {                         // stack with the top element removed
    return s / 10;
}

func peek(int s) {                           // top value, or -1 when empty
    return s % 10 - 1;
}

func isEmpty(int s) {
    return s == 0;
}

func size(int s) {
    int n = 0;
    while (s > 0) {
        s = s / 10;
        n++;
    }
    return n;
}

func sumStack(int s) {
    int sum = 0;
    while (s > 0) {
        sum = sum + peek(s);
        s = popTop(s);
    }
    return sum;
}

func maxStack(int s) {
    int mx = 0 - 1;
    while (s > 0) {
        int top = s % 10 - 1;
        if (top > mx) { mx = top; }
        s = s / 10;
    }
    return mx;
}

func containsStack(int s, int val) {
    while (s > 0) {
        if (s % 10 - 1 == val) { return 1; }
        s = s / 10;
    }
    return 0;
}

func countOccur(int s, int val) {
    int n = 0;
    while (s > 0) {
        if (s % 10 - 1 == val) { n++; }
        s = s / 10;
    }
    return n;
}

func sortStack(int s) {                       // counting sort, ascending from the bottom
    int result = 0;
    for (int v = 0; v < 9; v++) {
        int cnt = countOccur(s, v);
        for (int k = 0; k < cnt; k++) {
            result = push(result, v);
        }
    }
    return result;
}

func reverseStack(int s) {                    // pop from s, push onto r -> order reversed
    int r = 0;
    while (s > 0) {
        r = push(r, peek(s));
        s = popTop(s);
    }
    return r;
}

func matA(int i, int j) {                     // [[1,2],[3,4]]
    if (i == 0 && j == 0) { return 1; }
    if (i == 0 && j == 1) { return 2; }
    if (i == 1 && j == 0) { return 3; }
    return 4;
}

func matB(int i, int j) {                     // [[5,6],[7,8]]
    if (i == 0 && j == 0) { return 5; }
    if (i == 0 && j == 1) { return 6; }
    if (i == 1 && j == 0) { return 7; }
    return 8;
}

func matMul(int i, int j) {                   // C = A*B, C[i][j] = sum_k A[i][k]*B[k][j]
    int sum = 0;
    for (int k = 0; k < 2; k++) {
        sum = sum + matA(i, k) * matB(k, j);
    }
    return sum;
}

func bsearchSquare(int target, int hi) {      // find i in [0,hi] with i*i==target, else -1
    int lo = 0;
    while (lo <= hi) {
        int mid = (lo + hi) / 2;
        int v = mid * mid;
        if (v == target) {
            return mid;
        } else if (v < target) {
            lo = mid + 1;
        } else {
            hi = mid - 1;
        }
    }
    return 0 - 1;
}

func digitCount10(int n) {                    // base-10 digit count via do-while, n >= 0
    int c = 0;
    do {
        c++;
        n = n / 10;
    } while (n > 0);
    return c;
}

func printNum(int n) {                        // n >= 0, decimal
    if (n >= 10) { printNum(n / 10); }
    putchar(48 + n % 10);
    return 0;
}

func printBase(int n, int base) {             // n >= 0, base in 2..16
    if (n >= base) { printBase(n / base, base); }
    int d = n % base;
    if (d < 10) {
        putchar(48 + d);                      // '0'..'9'
    } else {
        putchar(55 + d);                      // 'A'..'F'  (55 + 10 == 'A')
    }
    return 0;
}

func printStack(int s) {                       // element values bottom..top, as digits
    if (s >= 10) { printStack(s / 10); }
    putchar(48 + (s % 10 - 1));
    return 0;
}

func main() {
    int failures = 0;

    // Build the stack [3,1,4,1,5] (bottom -> top).
    int s = 0;
    s = push(s, 3);
    s = push(s, 1);
    s = push(s, 4);
    s = push(s, 1);
    s = push(s, 5);

    // --- basic stack behaviour ---
    if (s != 42526)                              { failures++; }
    if (size(s) != 5)                            { failures++; }
    if (isEmpty(s) != 0 || isEmpty(0) != 1)      { failures++; }
    if (peek(s) != 5)                            { failures++; }
    if (peek(popTop(s)) != 1)                    { failures++; }
    if (size(popTop(s)) != 4)                    { failures++; }

    // --- aggregates ---
    if (sumStack(s) != 14)                       { failures++; }
    if (maxStack(s) != 5)                        { failures++; }
    if (maxStack(0) != -1)                       { failures++; }
    if (containsStack(s, 4) != 1)                { failures++; }
    if (containsStack(s, 3) != 1)                { failures++; }
    if (containsStack(s, 8) != 0)                { failures++; }
    if (countOccur(s, 1) != 2)                   { failures++; }
    if (countOccur(s, 9) != 0)                   { failures++; }

    // --- counting sort: [3,1,4,1,5] -> [1,1,3,4,5] ---
    int sorted = sortStack(s);
    if (sorted != 22456)                         { failures++; }
    if (size(sorted) != size(s))                 { failures++; }
    if (sumStack(sorted) != sumStack(s))         { failures++; }   // multiset preserved
    if (maxStack(sorted) != 5)                   { failures++; }
    if (peek(sorted) != 5)                        { failures++; }

    // --- reversal is its own inverse ---
    int rev = reverseStack(s);
    if (rev != 62524)                            { failures++; }
    if (reverseStack(rev) != s)                  { failures++; }

    // --- 2x2 matrix multiply, C = [[19,22],[43,50]] ---
    if (matMul(0, 0) != 19 || matMul(0, 1) != 22)  { failures++; }
    if (matMul(1, 0) != 43 || matMul(1, 1) != 50)  { failures++; }
    if (matMul(0, 0) + matMul(1, 1) != 69)       { failures++; }   // trace

    // --- binary search over f(i) = i*i ---
    if (bsearchSquare(49, 20) != 7)              { failures++; }
    if (bsearchSquare(144, 20) != 12)            { failures++; }
    if (bsearchSquare(0, 20) != 0)               { failures++; }
    if (bsearchSquare(400, 20) != 20)            { failures++; }
    if (bsearchSquare(50, 20) != -1)             { failures++; }

    // --- number formatting ---
    if (digitCount10(0) != 1 || digitCount10(255) != 3) { failures++; }
    if (digitCount10(1023) != 4)                 { failures++; }

    // --- deterministic output ---
    printStack(s);           putchar(10);        // "31415\n"
    printStack(sorted);      putchar(10);        // "11345\n"
    printNum(sumStack(s));   putchar(10);        // "14\n"
    printBase(255, 16);      putchar(10);        // "FF\n"
    printBase(1023, 16);     putchar(10);        // "3FF\n"
    printBase(10, 2);        putchar(10);        // "1010\n"

    if (failures == 0) {
        putchar(79); putchar(75); putchar(10);   // "OK\n"
    }
    return failures;
}
