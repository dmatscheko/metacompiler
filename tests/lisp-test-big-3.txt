; Lisp big test 3 - NUMBER THEORY & NUMBER FORMATTING.
;
; Digit manipulation, primality (trial division), prime counting and the nth prime,
; prime factorisation, divisor sums and perfect numbers, modular exponentiation,
; binomial coefficients, narcissistic (Armstrong) numbers, and a decimal formatter
; that prints integers one digit at a time with putchar - the only output the
; language offers. The printed report is fully deterministic, and the program still
; ends in 0 exactly when every numeric check passed.

; --- digit manipulation -------------------------------------------------------

(define (digit-sum n) (if (= n 0) 0 (+ (mod n 10) (digit-sum (/ n 10)))))
(define (digit-count n) (if (< n 10) 1 (+ 1 (digit-count (/ n 10)))))

(define (rev-digits-acc n acc)
  (if (= n 0) acc (rev-digits-acc (/ n 10) (+ (* acc 10) (mod n 10)))))
(define (rev-digits n) (rev-digits-acc n 0))
(define (palindrome? n) (= n (rev-digits n)))

; --- primality ----------------------------------------------------------------

(define (divides? d n) (= (mod n d) 0))

(define (prime-helper n i)
  (if (> (* i i) n)
      1
      (if (divides? i n) 0 (prime-helper n (+ i 1)))))
(define (prime? n) (if (< n 2) 0 (prime-helper n 2)))

(define (count-primes-h n i)
  (if (> i n) 0 (+ (prime? i) (count-primes-h n (+ i 1)))))
(define (count-primes n) (count-primes-h n 2))

(define (nth-prime-h k i)
  (if (prime? i)
      (if (= k 1) i (nth-prime-h (- k 1) (+ i 1)))
      (nth-prime-h k (+ i 1))))
(define (nth-prime k) (nth-prime-h k 2))

; --- factorisation ------------------------------------------------------------

(define (lpf-h n d)
  (if (> (* d d) n)
      n
      (if (divides? d n) (lpf-h (/ n d) d) (lpf-h n (+ d 1)))))
(define (largest-prime-factor n) (lpf-h n 2))

(define (div-out n d) (if (divides? d n) (div-out (/ n d) d) n))
(define (omega-h n d count)
  (if (> (* d d) n)
      (if (> n 1) (+ count 1) count)
      (if (divides? d n)
          (omega-h (div-out n d) (+ d 1) (+ count 1))
          (omega-h n (+ d 1) count))))
(define (omega n) (omega-h n 2 0))                 ; count of distinct prime factors

; --- divisors & perfect numbers -----------------------------------------------

(define (divsum-h n i acc)
  (if (>= i n)
      acc
      (if (divides? i n)
          (divsum-h n (+ i 1) (+ acc i))
          (divsum-h n (+ i 1) acc))))
(define (divsum n) (divsum-h n 1 0))               ; sum of proper divisors
(define (perfect? n) (= (divsum n) n))

; --- modular exponentiation ---------------------------------------------------

(define (even? n) (= (mod n 2) 0))
(define (modpow b e m)
  (if (= e 0)
      1
      (if (even? e)
          (let ((h (modpow b (/ e 2) m))) (mod (* h h) m))
          (mod (* b (modpow b (- e 1) m)) m))))

; --- factorials & binomials ---------------------------------------------------

(define (fact n) (if (< n 2) 1 (* n (fact (- n 1)))))
(define (choose n k) (/ (fact n) (* (fact k) (fact (- n k)))))

; --- Armstrong (narcissistic) numbers, 3 digit case ---------------------------

(define (cube x) (* x x x))
(define (arm-h n) (if (= n 0) 0 (+ (cube (mod n 10)) (arm-h (/ n 10)))))
(define (armstrong? n) (= (arm-h n) n))

; --- decimal formatter: print an integer digit by digit -----------------------

(define (print-digit d) (putchar (+ 48 d)))
(define (print-int-h n)
  (if (> n 0)
      (begin (print-int-h (/ n 10)) (print-digit (mod n 10)) 0)
      0))
(define (print-int n) (if (= n 0) (print-digit 0) (print-int-h n)))
(define (space) (putchar 32))
(define (newline) (putchar 10))

; print the first k primes on one line, space separated
(define (print-primes-h k i)
  (if (> k 0)
      (if (prime? i)
          (begin (print-int i) (space) (print-primes-h (- k 1) (+ i 1)))
          (print-primes-h k (+ i 1)))
      0))
(define (report)
  (begin
    (print-primes-h 8 2)                            ; 2 3 5 7 11 13 17 19
    (newline)
    (print-int (count-primes 100))                  ; 25
    (newline)
    0))

(define (check got want) (if (= got want) 0 1))

; --- deterministic printed report (its value is discarded) --------------------

(report)

; --- self check ---------------------------------------------------------------

(let ((failures
       (+ ; digits
          (check (digit-sum 12345) 15)
          (check (digit-sum 0) 0)
          (check (digit-count 0) 1)
          (check (digit-count 1000) 4)
          (check (rev-digits 1234) 4321)
          (check (palindrome? 12321) 1)
          (check (palindrome? 1234) 0)
          (check (palindrome? 1221) 1)
          ; primality
          (check (prime? 2) 1)
          (check (prime? 1) 0)
          (check (prime? 17) 1)
          (check (prime? 18) 0)
          (check (prime? 97) 1)
          (check (count-primes 10) 4)
          (check (count-primes 30) 10)
          (check (count-primes 100) 25)
          (check (nth-prime 1) 2)
          (check (nth-prime 5) 11)
          (check (nth-prime 10) 29)
          (check (nth-prime 25) 97)
          ; factorisation
          (check (largest-prime-factor 13195) 29)
          (check (largest-prime-factor 12) 3)
          (check (largest-prime-factor 97) 97)
          (check (omega 12) 2)
          (check (omega 30) 3)
          (check (omega 13195) 4)
          ; divisors & perfect numbers
          (check (divsum 6) 6)
          (check (divsum 12) 16)
          (check (perfect? 6) 1)
          (check (perfect? 28) 1)
          (check (perfect? 12) 0)
          (check (perfect? 496) 1)
          ; modular exponentiation
          (check (modpow 2 10 1000) 24)
          (check (modpow 3 3 7) 6)
          (check (modpow 2 0 5) 1)
          (check (modpow 10 5 9) 1)                 ; 10 == 1 (mod 9)
          ; binomials
          (check (fact 6) 720)
          (check (choose 5 2) 10)
          (check (choose 10 3) 120)
          (check (choose 12 6) 924)
          (check (choose 6 0) 1)
          ; Armstrong numbers
          (check (armstrong? 153) 1)
          (check (armstrong? 370) 1)
          (check (armstrong? 371) 1)
          (check (armstrong? 407) 1)
          (check (armstrong? 100) 0)
          ; the formatter agrees with the arithmetic
          (check (rev-digits (nth-prime 10)) 92))))  ; nth-prime 10 = 29 -> 92
  (if (= failures 0)
      (begin (putchar 79) (putchar 75) (putchar 10) 0)   ; print "OK\n"
      failures))
